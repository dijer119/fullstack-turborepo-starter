interface ElestockItem {
  rcept_no?: string;
  repror?: string;
  isu_exctv_ofcps?: string;
  isu_exctv_rgist_at?: string;
  isu_main_shrholdr?: string;
  stkrt_irds_resn?: string;
  trde_dd?: string;            // YYYYMMDD
  trde_qy?: string;            // 거래수량 (음수 가능)
  unit_pic?: string;           // 단가
  trde_amt?: string;
  posesn_stock_co_after?: string;
}

interface ElestockResponse {
  status?: string;
  message?: string;
  list?: ElestockItem[];
}

function sleep(ms: number): Promise<void> {
  return new Promise((r) => setTimeout(r, ms));
}

function isRetriable(err: unknown): boolean {
  if (!(err instanceof Error)) return false;
  const m = err.message.toLowerCase();
  return (
    m.includes("fetch failed") ||
    m.includes("timeout") ||
    m.includes("network") ||
    m.includes("econnreset") ||
    m.includes("etimedout")
  );
}

function parseSignedNum(raw: string | undefined): number | null {
  if (!raw) return null;
  const cleaned = raw.replace(/,/g, "").trim();
  if (cleaned === "" || cleaned === "-") return null;
  const n = Number(cleaned);
  return Number.isFinite(n) ? n : null;
}

function formatYmd(raw: string | undefined): string | null {
  if (!raw) return null;
  const digits = raw.replace(/[^\d]/g, "");
  if (digits.length < 8) return null;
  return `${digits.slice(0, 4)}-${digits.slice(4, 6)}-${digits.slice(6, 8)}`;
}

function derivePosition(item: ElestockItem): string | null {
  // 직위 (예: 상무, 대표이사). 비어있으면 등기 여부·주요주주 여부로 fallback.
  if (item.isu_exctv_ofcps) return item.isu_exctv_ofcps;
  if (item.isu_exctv_rgist_at === "Y") return "등기임원";
  if (item.isu_main_shrholdr === "Y") return "주요주주";
  return null;
}

export async function fetchElestock(params: {
  corpCode: string;
  bgnDe: string;
  endDe: string;
}): Promise<ElestockResponse | null> {
  const apiKey = process.env.DART_API_KEY;
  if (!apiKey) throw new Error("DART_API_KEY not set");
  const url = new URL("https://opendart.fss.or.kr/api/elestock.json");
  url.searchParams.set("crtfc_key", apiKey);
  url.searchParams.set("corp_code", params.corpCode);
  url.searchParams.set("bgn_de", params.bgnDe);
  url.searchParams.set("end_de", params.endDe);

  const MAX_ATTEMPTS = 3;
  for (let attempt = 1; attempt <= MAX_ATTEMPTS; attempt++) {
    try {
      const resp = await fetch(url, { signal: AbortSignal.timeout(30_000) });
      if (!resp.ok) return null;
      return (await resp.json()) as ElestockResponse;
    } catch (err) {
      if (attempt < MAX_ATTEMPTS && isRetriable(err)) {
        await sleep(500 * 2 ** (attempt - 1));
        continue;
      }
      console.warn(
        `[dart] elestock fetch failed corp=${params.corpCode}:`,
        err instanceof Error ? err.message : err,
      );
      return null;
    }
  }
  return null;
}

/** rcpNo → ElestockDetail Map. 응답 0건이거나 실패면 빈 Map. */
export async function buildElestockMap(params: {
  corpCode: string;
  bgnDe: string;
  endDe: string;
}): Promise<Map<string, ElestockDetail>> {
  const resp = await fetchElestock(params);
  const map = new Map<string, ElestockDetail>();
  if (!resp || resp.status !== "000" || !resp.list) return map;

  for (const item of resp.list) {
    if (!item.rcept_no) continue;
    const detail: ElestockDetail = {
      type: "elestock",
      filer: item.repror ?? null,
      position: derivePosition(item),
      reason: item.stkrt_irds_resn ?? null,
      changeDate: formatYmd(item.trde_dd),
      tradeQuantity: parseSignedNum(item.trde_qy),
      tradePrice: parseSignedNum(item.unit_pic),
      tradeAmount: parseSignedNum(item.trde_amt),
      afterShares: parseSignedNum(item.posesn_stock_co_after),
    };
    map.set(item.rcept_no, detail);
  }
  return map;
}

async function fetchOwnership(code: string, corpCode: string): Promise<FetchStats> {
  const { bgnDe, endDe } = todayDateRange();
  const resp = await fetchDartList({
    corpCode,
    pblntfTy: "D",
    bgnDe,
    endDe,
    pageCount: 100,
  });
  if (!resp || resp.status !== "000" || !resp.list) {
    return { inserted: 0, updated: 0, failed: 1 };
  }

  // 두 상세 API를 종목 단위로 한 번씩 호출해 rcpNo Map 구축.
  // 실패는 빈 Map으로 graceful degrade.

    buildElestockMap({ corpCode, bgnDe, endDe }),
    buildMajorstockMap({ corpCode, bgnDe, endDe }),
  ]);

  const rows = resp.list
    .slice(0, MAX_REPORTS_PER_STOCK)
    .map((it) => ({
      rcpNo: it.rcept_no,
      reportNm: it.report_nm,
      rceptDt: new Date(
        `${it.rcept_dt.slice(0, 4)}-${it.rcept_dt.slice(4, 6)}-${it.rcept_dt.slice(6, 8)}`,
      ),
    }));

  let inserted = 0;
  let updated = 0;
  for (const r of rows) {
    const base = classifyOwnership(r.reportNm);
    let detail: OwnershipDetail | null = null;
    if (base.reportType === "임원·주요주주") {
      detail = elestockMap.get(r.rcpNo) ?? null;
    } else if (base.reportType === "주식대량보유") {
      detail = majorMap.get(r.rcpNo) ?? null;
    }
    const payload = { ...base, detail };

    const existing = await db.disclosure.findUnique({ where: { rcpNo: r.rcpNo } });
    await db.disclosure.upsert({
      where: { rcpNo: r.rcpNo },
      create: {
        rcpNo: r.rcpNo,
        code,
        corpCode,
        reportNm: r.reportNm,
        pblntfTy: "D",
        rceptDt: r.rceptDt,
        category: "지분",
        payload: JSON.stringify(payload),
      },
      update: {
        payload: JSON.stringify(payload),
        fetchedAt: new Date(),
      },
    });
    if (existing) updated++;
    else inserted++;
  }
  return { inserted, updated, failed: 0 };
}

function formatRceptDt(d: Date): string {
  const yy = d.getFullYear();
  const mm = String(d.getMonth() + 1).padStart(2, "0");
  const dd = String(d.getDate()).padStart(2, "0");
  const hh = String(d.getHours()).padStart(2, "0");
  const mi = String(d.getMinutes()).padStart(2, "0");
  return `${yy}.${mm}.${dd} ${hh}:${mi}`;
}

function formatYmd(s: string | null): string {
  if (!s) return "—";
  return s.replace(/-/g, ".");
}

function formatShares(v: number | null): string {
  if (v == null) return "—";
  const sign = v > 0 ? "+" : v < 0 ? "" : "";   // 음수는 자체 부호 포함
  return `${sign}${v.toLocaleString()}주`;
}

function formatPrice(v: number | null): string {
  if (v == null) return "—";
  return `${v.toLocaleString()}원`;
}

function formatAmount(v: number | null): string {
  if (v == null) return "—";
  const abs = Math.abs(v);
  if (abs >= 1e8) return `${(v / 1e8).toFixed(2)}억`;
  if (abs >= 1e4) return `${Math.round(v / 1e4).toLocaleString()}만원`;
  return `${v.toLocaleString()}원`;
}

function quantityColor(v: number | null): string {
  if (v == null) return "text-gray-300";
  if (v > 0) return "text-green-300";
  if (v < 0) return "text-red-300";
  return "text-gray-300";
}

function ElestockBody({ d }: { d: ElestockDetail }) {
  return (
    <div className="mt-3 space-y-2 text-sm">
      <div className="flex flex-wrap gap-x-4">
        <span>보고자: <b>{d.filer ?? "—"}</b></span>
        <span>직위: <b>{d.position ?? "—"}</b></span>
      </div>
      <div>보고사유: <b>{d.reason ?? "—"}</b></div>
      <div className="border-t border-amber-700/40 pt-2 space-y-1">
        <div>변동일: <b>{formatYmd(d.changeDate)}</b></div>
        <div>
          거래수량: <b className={`font-mono ${quantityColor(d.tradeQuantity)}`}>{formatShares(d.tradeQuantity)}</b>
        </div>
        <div>거래가격: <b>{formatPrice(d.tradePrice)}</b></div>
        <div>거래금액: <b>{formatAmount(d.tradeAmount)}</b></div>
        <div>거래후보유주식수: <b>{d.afterShares != null ? `${d.afterShares.toLocaleString()}주` : "—"}</b></div>
      </div>

}
