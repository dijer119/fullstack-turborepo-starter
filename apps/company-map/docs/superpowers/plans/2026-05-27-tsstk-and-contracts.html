interface ContractItem {
  rcept_no?: string;
  ctr_dd?: string;          // 계약일 YYYYMMDD
  ctr_prd_bgn_dd?: string;  // 계약기간 시작
  ctr_prd_end_dd?: string;  // 계약기간 종료
  ctr_cn?: string;          // 계약 내용
  cn?: string;
  cnt_pa_nm?: string;       // 상대방
  ctr_pa_nm?: string;
  spy_aro_nm?: string;      // 공급지역
  regn?: string;
  spy_amt?: string;         // 계약금액
  ctr_amt?: string;
  rce_sl_amt_rat?: string;  // 최근매출액 대비 %
  [k: string]: string | undefined;
}

interface ContractResponse {
  status?: string;
  message?: string;
  list?: ContractItem[];
}

function sleep(ms: number): Promise<void> {
  return new Promise((r) => setTimeout(r, ms));
}

function isRetriable(err: unknown): boolean {
  if (!(err instanceof Error)) return false;
  const m = err.message.toLowerCase();
  return (
    m.includes("fetch failed") ||
    m.includes("timeout") ||
    m.includes("network") ||
    m.includes("econnreset") ||
    m.includes("etimedout")
  );
}

function parseNum(raw: string | undefined): number | null {
  if (!raw) return null;
  const cleaned = raw.replace(/,/g, "").trim();
  if (cleaned === "" || cleaned === "-") return null;
  const n = Number(cleaned);
  return Number.isFinite(n) ? n : null;
}

function formatYmd(raw: string | undefined): string | null {
  if (!raw) return null;
  const digits = raw.replace(/[^\d]/g, "");
  if (digits.length < 8) return null;
  return `${digits.slice(0, 4)}-${digits.slice(4, 6)}-${digits.slice(6, 8)}`;
}

export async function fetchContractSupply(params: {
  corpCode: string;
  bgnDe: string;
  endDe: string;
}): Promise<ContractResponse | null> {
  const apiKey = process.env.DART_API_KEY;
  if (!apiKey) throw new Error("DART_API_KEY not set");
  const url = new URL("https://opendart.fss.or.kr/api/cnctSlsCnclSDcsn.json");
  url.searchParams.set("crtfc_key", apiKey);
  url.searchParams.set("corp_code", params.corpCode);
  url.searchParams.set("bgn_de", params.bgnDe);
  url.searchParams.set("end_de", params.endDe);

  const MAX_ATTEMPTS = 3;
  for (let attempt = 1; attempt <= MAX_ATTEMPTS; attempt++) {
    try {
      const resp = await fetch(url, { signal: AbortSignal.timeout(30_000) });
      if (!resp.ok) return null;
      return (await resp.json()) as ContractResponse;
    } catch (err) {
      if (attempt < MAX_ATTEMPTS && isRetriable(err)) {
        await sleep(500 * 2 ** (attempt - 1));
        continue;
      }
      console.warn(
        `[dart] contract-supply fetch failed corp=${params.corpCode}:`,
        err instanceof Error ? err.message : err,
      );
      return null;
    }
  }
  return null;
}

export async function buildContractMap(params: {
  corpCode: string;
  bgnDe: string;
  endDe: string;
}): Promise<Map<string, ContractPayload>> {
  const resp = await fetchContractSupply(params);
  const map = new Map<string, ContractPayload>();
  if (!resp || resp.status !== "000" || !resp.list) return map;

  for (const item of resp.list) {
    if (!item.rcept_no) continue;
    const payload: ContractPayload = {
      contractDate: formatYmd(item.ctr_dd),
      startDate: formatYmd(item.ctr_prd_bgn_dd),
      endDate: formatYmd(item.ctr_prd_end_dd),
      contractContent: item.ctr_cn ?? item.cn ?? null,
      counterparty: item.cnt_pa_nm ?? item.ctr_pa_nm ?? null,
      supplyRegion: item.spy_aro_nm ?? item.regn ?? null,
      amount: parseNum(item.spy_amt ?? item.ctr_amt),
      recentSalesRatio: parseNum(item.rce_sl_amt_rat),
    };
    map.set(item.rcept_no, payload);
  }
  return map;
}

          `${it.rcept_dt.slice(0, 4)}-${it.rcept_dt.slice(4, 6)}-${it.rcept_dt.slice(6, 8)}`,
        ),
      });
    }

  // rceptDt desc 정렬 후 16건 제한
  rows.sort((a, b) => b.rceptDt.getTime() - a.rceptDt.getTime());
  const limited = rows.slice(0, MAX_REPORTS_PER_STOCK);

  let inserted = 0;
  let updated = 0;

        rceptDt: r.rceptDt,
        category: "지분",
        payload: JSON.stringify(payload),
      },
      update: {
        payload: JSON.stringify(payload),
        fetchedAt: new Date(),
      },
    });
    if (existing) updated++;
    else inserted++;
  }
  return { inserted, updated, failed: 0 };
}

function formatRceptDt(d: Date): string {
  const yy = d.getFullYear();
  const mm = String(d.getMonth() + 1).padStart(2, "0");
  const dd = String(d.getDate()).padStart(2, "0");
  const hh = String(d.getHours()).padStart(2, "0");
  const mi = String(d.getMinutes()).padStart(2, "0");
  return `${yy}.${mm}.${dd} ${hh}:${mi}`;
}

function formatYmd(s: string | null): string {
  if (!s) return "—";
  return s.replace(/-/g, ".");
}

function formatAmount(v: number | null): string {
  if (v == null) return "—";
  const abs = Math.abs(v);
  if (abs >= 1e12) return `${(v / 1e12).toFixed(1)}조`;
  if (abs >= 1e8) {
    const eok = Math.floor(v / 1e8);
    const remManwon = Math.floor((v - eok * 1e8) / 1e4);
    if (remManwon > 0) return `${eok.toLocaleString()}억 ${remManwon.toLocaleString()}만원`;
    return `${eok.toLocaleString()}억`;
  }
  if (abs >= 1e4) return `${Math.round(v / 1e4).toLocaleString()}만원`;
  return `${v.toLocaleString()}원`;
}

function Row({ label, value }: { label: string; value: React.ReactNode }) {
  if (value == null || value === "" || value === "—") return null;
  return (
    <div>
      {label}: <b>{value}</b>
    </div>
  );
}

export function ContractCard({ reportNm, rceptDt, payload }: Props) {
  const p = payload;
  const period =
    p?.startDate && p.endDate
      ? `${formatYmd(p.startDate)} - ${formatYmd(p.endDate)}`
      : null;

  return (
    <article className="rounded-lg bg-cyan-900/80 p-5 text-gray-100 shadow">
      <h3 className="text-lg font-bold">{reportNm}</h3>
      {p && (
        <div className="mt-3 space-y-1 text-sm">
          <Row label="계약일" value={p.contractDate ? formatYmd(p.contractDate) : null} />
          <Row label="계약기간" value={period} />
          <Row label="계약내용" value={p.contractContent} />
          <Row label="계약상대방" value={p.counterparty} />
          <Row label="공급지역" value={p.supplyRegion} />
          <Row label="계약금액" value={p.amount != null ? formatAmount(p.amount) : null} />
          <Row
            label="최근매출액대비"
            value={p.recentSalesRatio != null ? `${p.recentSalesRatio.toFixed(2)}%` : null}
          />
        </div>
      )}
      <p className="mt-4 text-xs text-gray-300">{formatRceptDt(rceptDt)}</p>
    </article>
  );
}
